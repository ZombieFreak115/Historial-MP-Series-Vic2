country_event = {
	id = 1000002
	title = "Host swap"
	desc = "Host's bout to swap yo ur nation"
	picture = "greatpowers"
	is_triggered_only = yes
	
	immediate = {
		FROM = { country_event = 1000003 }
		random_country = {
			limit = {
				has_global_flag = host_flag_tagmove
			}
			UTI = { country_event = 1000004 }
		}
		random_country = {
			limit = {
				NOT = { has_global_flag = host_flag_tagmove }
			}
			UTI = { country_event = 1000005 }
		}
	}
	
	
	option = {
		name = "k"
	}
}
