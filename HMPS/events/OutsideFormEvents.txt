country_event = {
	id = 5329627
	title = "Formation of Italy"
	desc = "EVTDESC5329627"
	is_triggered_only = yes
	
	option = {
		name = "Northern Italians"
		any_country = {
			limit = {
				OR = {
					primary_culture = north_italian
					primary_culture = south_italian
				}
				in_sphere = THIS
			}
			annex_to = THIS
		}
		any_owned = {
			limit = {
				is_core = THIS
				region = ITA_1
			}
			remove_core = ITA
		}
		release_vassal = ITA
		create_alliance = ITA
		diplomatic_influence = {
			who = ITA
			value = 300
		}
	}
	
	option = {
		name = "Southern Italians"
		any_country = {
			limit = {
				OR = {
					primary_culture = north_italian
					primary_culture = south_italian
				}
				in_sphere = THIS
			}
			annex_to = THIS
		}
		any_owned = {
			limit = {
				is_core = THIS
				region = ITA_1
			}
			remove_core = ITA
		}
		release_vassal = ITA
		ITA = {
			remove_accepted_culture = south_italian
			primary_culture = south_italian
			add_accepted_culture = north_italian
		}
		create_alliance = ITA
		diplomatic_influence = {
			who = ITA
			value = 300
		}
	}
}
